Create /src/services/imageProcessor.ts

Use browser Canvas API — no native libraries needed.

processImage(file: File, options: {
  quality: 'compress' | 'original'
  maxSizeMB: number | null
}): Promise<{
  mainBlob: Blob
  thumbnailBlob: Blob
  finalSizeBytes: number
}>

Steps:
1. Validate file type: must be image/jpeg or image/png
   If invalid → throw new Error('INVALID_FORMAT')

2. Check file size:
   If maxSizeMB != null AND file.size > maxSizeMB * 1024 * 1024:
     throw new Error('IMAGE_TOO_LARGE')

3. Load image into HTMLImageElement:
   const img = new Image()
   const objectUrl = URL.createObjectURL(file)
   await new Promise: img.onload = resolve, img.onerror = reject
   img.src = objectUrl
   URL.revokeObjectURL(objectUrl)

4. Main image (max width 1200px, maintain aspect ratio):
   const canvas = document.createElement('canvas')
   const scale = Math.min(1, 1200 / img.width)
   canvas.width = img.width * scale
   canvas.height = img.height * scale
   ctx.drawImage(img, 0, 0, canvas.width, canvas.height)

   If quality = 'compress':
     canvas.toBlob(resolve, 'image/jpeg', 0.7)
   If quality = 'original':
     canvas.toBlob(resolve, 'image/jpeg', 1.0)

5. Thumbnail (max width 300px):
   const thumbCanvas = document.createElement('canvas')
   const thumbScale = Math.min(1, 300 / img.width)
   thumbCanvas.width = img.width * thumbScale
   thumbCanvas.height = img.height * thumbScale
   ctx.drawImage(img, 0, 0, thumbCanvas.width, thumbCanvas.height)
   thumbCanvas.toBlob(resolve, 'image/jpeg', 0.6)

6. Return { mainBlob, thumbnailBlob, finalSizeBytes: mainBlob.size }

blobToBase64(blob: Blob): Promise<string>
  Use FileReader to convert blob → base64 string
  Strip the data:...;base64, prefix — return only the base64 data

Create /src/services/fileSystemService.ts

Using @capacitor/filesystem — Directory.Data

All paths are relative to Directory.Data + 'medora/'

saveFile(options: {
  relativePath: string   // e.g. 'cases/abc123/images/img_1234.jpg'
  base64Data: string
  mimeType?: string
}): Promise<void>

  Filesystem.writeFile({
    path: 'medora/' + relativePath,
    data: base64Data,
    directory: Directory.Data,
    recursive: true,
  })

readFile(relativePath: string): Promise<string>
  Returns base64 string of file contents
  Filesystem.readFile({ path: 'medora/' + relativePath, directory: Directory.Data })

deleteFile(relativePath: string): Promise<void>
  Filesystem.deleteFile({ path: 'medora/' + relativePath, directory: Directory.Data })

fileExists(relativePath: string): Promise<boolean>
  Try Filesystem.stat() → return true
  Catch → return false

getFileUrl(relativePath: string): string
  Return: `capacitor://localhost/_capacitor_file_/medora/${relativePath}`
  This URL can be used directly in <img src="...">

getFolderSize(folderPath: string): Promise<number>
  Recursively list all files in folder
  Sum their sizes
  Return total bytes

Create /src/services/mediaService.ts

Manages media attached to cases.

getByCaseId(caseId: string): Promise<{
  id: string
  filePath: string
  thumbnailPath: string
  fileUrl: string
  thumbnailUrl: string
  fileSize: number
}[]>

  SELECT * FROM media WHERE case_id = ? AND is_deleted = 0
  ORDER BY created_at DESC
  For each: add fileUrl and thumbnailUrl using fileSystemService.getFileUrl()

add(caseId: string, file: File): Promise<{ id: string, fileUrl: string, thumbnailUrl: string }>

  1. Read settings: imageQuality, imageMaxSize
  2. Process: imageProcessor.processImage(file, { quality, maxSizeMB })
  3. Generate paths:
     const id = generateId()
     const timestamp = Date.now()
     mainPath = `cases/${caseId}/images/${id}_${timestamp}.jpg`
     thumbPath = `cases/${caseId}/thumbnails/${id}_${timestamp}_thumb.jpg`
  4. Convert blobs to base64 via blobToBase64()
  5. Save both files via fileSystemService.saveFile()
  6. INSERT INTO media (id, case_id, file_path, thumbnail_path, file_size)
  7. enqueueSyncJob('media', id, 'INSERT')
  8. Return { id, fileUrl, thumbnailUrl }

delete(mediaId: string): Promise<void>
  SELECT file_path, thumbnail_path FROM media WHERE id = ?
  fileSystemService.deleteFile(file_path)
  fileSystemService.deleteFile(thumbnail_path)
  UPDATE media SET is_deleted = 1 WHERE id = ?
  enqueueSyncJob('media', mediaId, 'DELETE')

deleteAllForCase(caseId: string): Promise<void>
  SELECT all media for case
  Delete all files
  UPDATE media SET is_deleted = 1 WHERE case_id = ?
