IMPORTANT: The tokenMap MUST be a local variable — never stored
in database, localStorage, or any persistent storage.

deIdentifyCase(
  caseData: { case: Case, investigations: Investigation[], management: ManagementEntry[], progressNotes: ProgressNote[] },
  tokenMap: Map<string, string>
): DeIdentifiedCase

  Token assignment:
    If patientId NOT in tokenMap:
      Assign next letter: 'Patient A', 'Patient B', ...
      tokenMap.set(patientId, 'Patient A')
    token = tokenMap.get(patientId)

  Date conversion (relative to admissionDate):
    admissionDate = Day 0 reference
    Any other date → calculate difference in days → 'Day X'
    Example: investigation date 2 days after admission → 'Day 2'

  Fields to INCLUDE:
    token, age, gender, specialty
    provisionalDiagnosis, chiefComplaint, presentHistory
    pastMedicalHistory, allergies, currentMedications
    investigations (name, type, result, relativeDate)
    management (type, medications?, mode?, details?, relativeDate)
    progressNotes (assessment, vitals, relativeDate)
    status, dischargeOutcome

  Fields to EXCLUDE (NEVER sent to AI):
    patientId, name, fileNumber, hospital, hospitalId
    exact dates, dobDay, dobMonth, dobYear
    any Google account info, API keys

deIdentifyMultipleCases(cases: {...}[]): DeIdentifiedCase[]

  const tokenMap = new Map<string, string>()  // fresh every call
  return cases.map(c => deIdentifyCase(c, tokenMap))
  // tokenMap is local → garbage collected after return

Create /src/services/aiProviderService.ts

sendRequest(prompt: string, systemPrompt: string): Promise<string>

1. Read from settingsService: aiProvider, apiKey, aiModel

2. If apiKey is empty or whitespace:
   throw new Error('NO_API_KEY')

3. If aiProvider = 'anthropic':
   POST https://api.anthropic.com/v1/messages
   Headers:
     'x-api-key': apiKey
     'anthropic-version': '2023-06-01'
     'content-type': 'application/json'
   Body:
     model:
       sonnet → 'claude-sonnet-4-5'
       opus   → 'claude-opus-4-5'
       haiku  → 'claude-haiku-4-5-20251001'
     max_tokens: 2000
     system: systemPrompt
     messages: [{ role: 'user', content: prompt }]
   Parse: response.content[0].text

4. If aiProvider = 'openai':
   POST https://api.openai.com/v1/chat/completions
   Headers:
     'Authorization': `Bearer ${apiKey}`
     'content-type': 'application/json'
   Body:
     model: 'gpt-4o'
     messages: [
       { role: 'system', content: systemPrompt },
       { role: 'user', content: prompt }
     ]
   Parse: response.choices[0].message.content

5. Error handling:
   response.status === 401 → throw new Error('INVALID_API_KEY')
   response.status === 429 → throw new Error('QUOTA_EXCEEDED')
   fetch timeout (30s) using AbortController → throw new Error('AI_TIMEOUT')
   catch network error → throw new Error('AI_NETWORK_ERROR')

Create /src/services/insightsService.ts

generateInsights(): Promise<{
  groups: {
    status: 'red' | 'yellow' | 'green'
    label: string
    cases: { name: string, summary: string, recommendation: string }[]
  }[]
}>

Steps:
1. caseService.getActiveForInsights()
   If empty → throw new Error('NO_ACTIVE_CASES')

2. deIdentificationService.deIdentifyMultipleCases(activeCases)

3. Get aiLanguage from settingsService

4. systemPrompt:
   "You are a clinical decision support assistant.
    Analyze the following de-identified patient cases
    and group them by clinical urgency.
    Respond in {aiLanguage === 'arabic' ? 'Arabic' : 'English'}.
    Return ONLY a valid JSON object with this exact structure:
    {
      \"groups\": [
        {
          \"status\": \"red\" | \"yellow\" | \"green\",
          \"label\": \"string\",
          \"cases\": [
            { \"name\": \"string\", \"summary\": \"string\", \"recommendation\": \"string\" }
          ]
        }
      ]
    }
    red = needs immediate attention
    yellow = requires review or plan adjustment
    green = stable, consider discharge
    Do not include any text outside the JSON."

5. userPrompt: JSON.stringify(deIdentifiedCases)

6. responseText = await aiProviderService.sendRequest(userPrompt, systemPrompt)

7. Parse:
   const clean = responseText.replace(/```json|```/g, '').trim()
   const parsed = JSON.parse(clean)
   If structure invalid → throw new Error('AI_INVALID_RESPONSE')

8. Return parsed.groups

Create /src/services/groupPearlService.ts

generateGroupPearl(filters: {
  diagnosis: string
  timePeriod: string
  outcome: string
  fromDate?: string
  toDate?: string
}): Promise<{
  summary: string
  patterns: string[]
  comparison: string
  pearl: string
  diseaseReview: string
}>

Steps:
1. Query matching cases from SQLite:
   SELECT * FROM cases JOIN patients WHERE
     provisional_diagnosis LIKE '%{diagnosis}%'
     AND is_deleted = 0
   Apply timePeriod filter on admission_date
   Apply outcome filter:
     'active'     → AND status = 'active'
     'discharged' → AND status = 'discharged'
     'died'       → AND discharge_outcome LIKE '%died%' OR '%death%' OR '%expired%'
     'all'        → no filter

2. If no cases → throw new Error('NO_MATCHING_CASES')

3. Fetch full data for each case (investigations, management, progressNotes)
4. deIdentificationService.deIdentifyMultipleCases(cases)
5. Get aiLanguage from settingsService

6. systemPrompt:
   "You are a clinical education assistant.
    Analyze this group of de-identified patient cases
    with the diagnosis: {diagnosis}.
    Provide clinical patterns, comparisons, and a learning pearl.
    Respond in {language}.
    Return ONLY a valid JSON object:
    {
      \"summary\": \"string\",
      \"patterns\": [\"string\"],
      \"comparison\": \"string\",
      \"pearl\": \"string\",
      \"diseaseReview\": \"string\"
    }
    Do not include any text outside the JSON."

7. userPrompt: JSON.stringify({ diagnosis, caseCount: cases.length, cases: deIdentifiedCases })
8. Send → parse → validate → return

Time period filter helper:
  month    → >= date(now, '-30 days')
  3months  → >= date(now, '-90 days')
  6months  → >= date(now, '-180 days')
  year     → >= date(now, '-365 days')
  custom   → between fromDate and toDate
  all      → no filter

Export 3 hooks:

─── useInsights() ─────────────────────────────────────────────

const [state, setState] = useState<'ready'|'loading'|'done'|'error'>('ready')
const [data, setData] = useState(null)
const [error, setError] = useState<string|null>(null)

generate = async () => {
  setState('loading'); setError(null)
  try {
    const result = await insightsService.generateInsights()
    setData(result); setState('done')
  } catch (e) {
    setError(mapAIError(e.message)); setState('error')
  }
}

reset = () => { setState('ready'); setData(null); setError(null) }
return { state, data, error, generate, reset }

─── useCasePearl(caseId) ──────────────────────────────────────

const [isLoading, setIsLoading] = useState(false)
const [data, setData] = useState(null)
const [error, setError] = useState<string|null>(null)

generate = async () => {
  setIsLoading(true); setError(null)
  try {
    const result = await casePearlService.generateCasePearl(caseId)
    setData(result)
  } catch (e) {
    setError(mapAIError(e.message))
  } finally { setIsLoading(false) }
}

return { isLoading, data, error, generate }

─── useGroupPearl() ───────────────────────────────────────────

const [state, setState] = useState<'empty'|'loading'|'results'|'error'|'no_cases'>('empty')
const [data, setData] = useState(null)
const [error, setError] = useState<string|null>(null)

generate = async (filters) => {
  setState('loading'); setError(null)
  try {
    const result = await groupPearlService.generateGroupPearl(filters)
    setData(result); setState('results')
  } catch (e) {
    if (e.message === 'NO_MATCHING_CASES') setState('no_cases')
    else { setError(mapAIError(e.message)); setState('error') }
  }
}

reset = () => { setState('empty'); setData(null); setError(null) }
return { state, data, error, generate, reset }

─── mapAIError(code) helper ───────────────────────────────────

'NO_API_KEY'          → 'No API key set. Go to Settings → AI Integration.'
'INVALID_API_KEY'     → 'Invalid API key. Go to Settings to update it.'
'QUOTA_EXCEEDED'      → 'AI quota exceeded. Check your API plan.'
'AI_TIMEOUT'          → 'Request timed out. Please try again.'
'AI_NETWORK_ERROR'    → 'Connection failed. Check your internet.'
'AI_INVALID_RESPONSE' → 'Unexpected AI response. Please try again.'
'NO_ACTIVE_CASES'     → 'No active cases to analyze.'
default               → 'An error occurred. Please try again.'
